fd = <network call>
filename = place we wanna write this stuff
<class>buffer buf;

while(ssize_t status = buf.read(fd) > 0):
    HTMLParser(buf.begin()) h;
    buf.write(filename) // Write what we got then clear buffer to re use 
    addToURLStuff(h.urlStuff)

if status is -1 we die (not really we just don't do the write (or maybe we do cuz
we could write partially on a fail potentially))

if status is 0 (we finished read of all HTML):
    buf.write(filename) // Last write
